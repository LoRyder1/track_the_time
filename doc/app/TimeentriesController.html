def stop_timer
        timee = Timeentry.find(params[:id])

        if timee.duration == nil
                timee.duration = Time.now - (timee.start_time)
        else
                timee.duration = timee.duration + (Time.now - (timee.start_time))
        end

        timee.start_time = 0
        timee.save
        redirect_to timeentries_path
end

def update
        @timee = Timeentry.find(params[:id])

        @timee.save
        redirect_to timeentries_path
end
